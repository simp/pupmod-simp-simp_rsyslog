class simp_rsyslog::local (
  String $order = 'HH'
){
  assert_private()

  rsyslog::rule::local { "${order}_simp_rsyslog_profile_local":
    rule            => $::simp_rsyslog::security_relevant_logs,
    target_log_file => $::simp_rsyslog::local_target,
    stop_processing => true
  }
}
